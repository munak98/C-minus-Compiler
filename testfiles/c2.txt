set union(set a, set b) {
    set result;
    elem x;
    result = EMPTY;
    forall(x in a) add(x in result);
    forall(x in b) add(x in result);
    return result;
}


set diff(set a, set b) {
    set result;
    elem x;
    result = EMPTY;
    forall(x in a) remove(x in b);
    return result;
}

    set first;
    set second;
    first = EMPTY;

    add((5+2-1) in add((1*2-3) in first));
    add((6*1/2) in add((1+6/2) in second));
