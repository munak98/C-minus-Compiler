int groupUnderAdd(set s) {
    elem x, y;
    forall (x in s) {
      forall (y in s) add((x+y) in s)
    }
    forall (k v) add((0-x) in s);
    @add(0 in s);
    return 0;
}

int groupUnderMult(set s;) {
    elem y, x;
    forall (x in s) {
      forall (y in s) add((x*y) in s);
    }
    forall (x in s) add((1/x) in s);
    add(1 in s);
    return 0;
}
